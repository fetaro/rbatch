def load()
  begin
    @yaml = YAML::load_file(@path)
  rescue
    # when run_conf does not exist, do nothing.
    @yaml = false
  end
  if @yaml
    @yaml.each_key do |key|
      if @@def_opt.has_key?(key.to_sym)
        @opt[key.to_sym]=@yaml[key]
      else
        raise RBatch::RunConfException, "\"#{key}\" is not available option"
      end
    end
  end
end

def merge(opt)
  tmp = @opt.clone
  opt.each_key do |key|
    if tmp.has_key?(key)
      tmp[key] = opt[key]
    else
      raise RBatch::RunConfException, "\"#{key}\" is not available option"
    end
  end
  return tmp
end
