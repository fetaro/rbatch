def initialize(verbose=nil)
  if ! verbose.nil?
    @journal_verbose = verbose
  else
    @journal_verbose = @@def_vars[:rbatch_journal_level]
  end
  @journals = []
  @user_logs = []
end

def add_log(log)
  @user_logs << log
  if @@def_vars[:mix_rbatch_journal_to_logs]
    @journals.each do |j|
      log.journal(j)
    end
  end
end

def put(level,str)
  if level <= @journal_verbose
    @journals << str
    str = "[RBatch] " + str
    puts str
    @user_logs.each do |log|
      if @@def_vars[:mix_rbatch_journal_to_logs]
        log.journal(str)
      end
    end
  end
end
