def initialize(path)
  @path = path
  begin
    @hash = ConfigElement.new(YAML::load_file(@path))
  rescue Errno::ENOENT => e
    @hash = nil
  end
end

def[](key)
  if @hash.nil?

  else
    @hash[key]
  end
end
