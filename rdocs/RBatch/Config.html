RBatch::Config

Read config file and return hash opject.

Default config file path is “../conf/(Program Base name).yaml“

Sample

config : ./conf/sample2.yaml 

 key: value
 array:
  * item1
  * item2
  * item3

script : ./bin/sample2.rb 

 require 'rbatch'
 p RBatch::Config.new
 # or  p RBatch::config
 => {"key" => "value", "array" => ["item1", "item2", "item3"]}
